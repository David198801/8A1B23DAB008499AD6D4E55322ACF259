业务类型

```javascript
SELECT
lpad('--',(level-1)*2,'--')||t.FNAME "层级关系",
CASE FLEAF WHEN 1 THEN t.FNAME END AS "划款类型",
t.FCODE ,t.FPARENT_CODE FROM T_PA_BUS_TYPE t
WHERE t.fdeleted=0 AND t.FCHECKED=1
START WITH t.FNAME='存款投资业务'
CONNECT BY t.fparent_code = PRIOR t.FCODE ;
```



```javascript
SELECT CASE t.FLEVEL WHEN 1 THEN t.FNAME WHEN 2 THEN '--'||t.FNAME END,t.* FROM T_PA_PRO_TYPE t CONNECT BY t.FPARENT_ID =PRIOR t.FID;
```



```javascript
SELECT
lpad('--',(level-1)*2,'--')||t.FNAME "层级关系",
CASE t.FLEAF WHEN 1 THEN t.fname END "产品类型",
t.FCODE ,t.FLEVEL ,t.FLEAF
FROM
(SELECT * FROM T_PA_PRO_TYPE WHERE FDELETED =0 AND FCHECKED =1) t
START WITH (t.FPARENT_ID IS NULL OR t.FPARENT_ID=-1)
CONNECT BY
t.FPARENT_ID = PRIOR t.FID;
```
